144 typedef struct {
145  int ie_status;
146  int in_critical;
147 } mxc_crit_state_t;
148 
149 static mxc_crit_state_t _state = { .ie_status = (int)0xFFFFFFFF, .in_critical = 0 };
150 
151 static inline void _mxc_crit_get_state(void)
152 {
153 #ifndef __riscv
154  /*
155  On ARM M the 0th bit of the Priority Mask register indicates
156  whether interrupts are enabled or not.
157 
158  0 = enabled
159  1 = disabled
160  */
161  uint32_t primask = __get_PRIMASK();
162  _state.ie_status = (primask == 0);
163 #else
164  /*
165  On RISC-V bit position 3 (Machine Interrupt Enable) of the
166  mstatus register indicates whether interrupts are enabled.
167 
168  0 = disabled
169  1 = enabled
170  */
171  uint32_t mstatus = get_mstatus();
172  _state.ie_status = ((mstatus & (1 << 3)) != 0);
173 #endif
174 }
175 
187 static inline void MXC_SYS_Crit_Enter(void)
188 {
189  _mxc_crit_get_state();
190  if (_state.ie_status)
191  __disable_irq();
192  _state.in_critical = 1;
193 }
194 
199 static inline void MXC_SYS_Crit_Exit(void)
200 {
201  if (_state.ie_status) {
202  __enable_irq();
203  }
204  _state.in_critical = 0;
205  _mxc_crit_get_state();
206  /*
207  ^ Reset the state again to prevent edge case
208  where interrupts get disabled, then Crit_Exit() gets
209  called, which would inadvertently re-enable interrupts
210  from old state.
211  */
212 }
213 
219 static inline int MXC_SYS_In_Crit_Section(void)
220 {
221  return _state.in_critical;
222 }
