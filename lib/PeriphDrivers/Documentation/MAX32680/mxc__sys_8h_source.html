195 typedef struct {
196  int ie_status;
197  int in_critical;
198 } mxc_crit_state_t;
199 
200 static mxc_crit_state_t _state = { .ie_status = (int)0xFFFFFFFF, .in_critical = 0 };
201 
202 static inline void _mxc_crit_get_state()
203 {
204 #ifndef __riscv
205  /*
206  On ARM M the 0th bit of the Priority Mask register indicates
207  whether interrupts are enabled or not.
208 
209  0 = enabled
210  1 = disabled
211  */
212  uint32_t primask = __get_PRIMASK();
213  _state.ie_status = (primask == 0);
214 #else
215  /*
216  On RISC-V bit position 3 (Machine Interrupt Enable) of the
217  mstatus register indicates whether interrupts are enabled.
218 
219  0 = disabled
220  1 = enabled
221  */
222  uint32_t mstatus = get_mstatus();
223  _state.ie_status = ((mstatus & (1 << 3)) != 0);
224 #endif
225 }
226 
238 static inline void MXC_SYS_Crit_Enter(void)
239 {
240  _mxc_crit_get_state();
241  if (_state.ie_status)
242  __disable_irq();
243  _state.in_critical = 1;
244 }
245 
250 static inline void MXC_SYS_Crit_Exit(void)
251 {
252  if (_state.ie_status) {
253  __enable_irq();
254  }
255  _state.in_critical = 0;
256  _mxc_crit_get_state();
257  /*
258  ^ Reset the state again to prevent edge case
259  where interrupts get disabled, then Crit_Exit() gets
260  called, which would inadvertently re-enable interrupts
261  from old state.
262  */
263 }
264 
270 static inline int MXC_SYS_In_Crit_Section(void)
271 {
272  return _state.in_critical;
273 }
