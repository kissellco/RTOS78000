233 typedef struct {
234  int ie_status;
235  int in_critical;
236 } mxc_crit_state_t;
237 
238 static mxc_crit_state_t _state = { .ie_status = (int)0xFFFFFFFF, .in_critical = 0 };
239 
240 static inline void _mxc_crit_get_state(void)
241 {
242 #ifndef __riscv
243  /*
244  On ARM M the 0th bit of the Priority Mask register indicates
245  whether interrupts are enabled or not.
246 
247  0 = enabled
248  1 = disabled
249  */
250  uint32_t primask = __get_PRIMASK();
251  _state.ie_status = (primask == 0);
252 #else
253  /*
254  On RISC-V bit position 3 (Machine Interrupt Enable) of the
255  mstatus register indicates whether interrupts are enabled.
256 
257  0 = disabled
258  1 = enabled
259  */
260  uint32_t mstatus = get_mstatus();
261  _state.ie_status = ((mstatus & (1 << 3)) != 0);
262 #endif
263 }
264 
276 static inline void MXC_SYS_Crit_Enter(void)
277 {
278  _mxc_crit_get_state();
279  if (_state.ie_status)
280  __disable_irq();
281  _state.in_critical = 1;
282 }
283 
288 static inline void MXC_SYS_Crit_Exit(void)
289 {
290  if (_state.ie_status) {
291  __enable_irq();
292  }
293  _state.in_critical = 0;
294  _mxc_crit_get_state();
295  /*
296  ^ Reset the state again to prevent edge case
297  where interrupts get disabled, then Crit_Exit() gets
298  called, which would inadvertently re-enable interrupts
299  from old state.
300  */
301 }
302 
308 static inline int MXC_SYS_In_Crit_Section(void)
309 {
310  return _state.in_critical;
311 }
