229 static mxc_crit_state_t _state = { .ie_status = (int)0xFFFFFFFF, .in_critical = 0 };
230 
231 static inline void _mxc_crit_get_state(void)
232 {
233 #ifndef __riscv
234  /*
235  On ARM M the 0th bit of the Priority Mask register indicates
236  whether interrupts are enabled or not.
237 
238  0 = enabled
239  1 = disabled
240  */
241  uint32_t primask = __get_PRIMASK();
242  _state.ie_status = (primask == 0);
243 #else
244  /*
245  On RISC-V bit position 3 (Machine Interrupt Enable) of the
246  mstatus register indicates whether interrupts are enabled.
247 
248  0 = disabled
249  1 = enabled
250  */
251  uint32_t mstatus = get_mstatus();
252  _state.ie_status = ((mstatus & (1 << 3)) != 0);
253 #endif
254 }
255 
267 static inline void MXC_SYS_Crit_Enter(void)
268 {
269  _mxc_crit_get_state();
270  if (_state.ie_status)
271  __disable_irq();
272  _state.in_critical = 1;
273 }
274 
279 static inline void MXC_SYS_Crit_Exit(void)
280 {
281  if (_state.ie_status) {
282  __enable_irq();
283  }
284  _state.in_critical = 0;
285  _mxc_crit_get_state();
286  /*
287  ^ Reset the state again to prevent edge case
288  where interrupts get disabled, then Crit_Exit() gets
289  called, which would inadvertently re-enable interrupts
290  from old state.
291  */
292 }
293 
299 static inline int MXC_SYS_In_Crit_Section(void)
300 {
301  return _state.in_critical;
302 }
